```python
from pydantic import BaseModel, ConfigDict, Field

class MonthlyStatPoint(BaseModel):
    month: int = Field(..., ge=1, le=12, description="Month number (1=Jan, 12=Dec)")
    median_price: float = Field(..., description="Median modal price (Rs/quintal)")
    q1_price: float = Field(..., description="25th percentile price")
    q3_price: float = Field(..., description="75th percentile price")
    iqr_price: float = Field(..., description="Interquartile range (q3 - q1)")
    record_count: int = Field(..., description="Number of daily price observations in this month")
    years_of_data: int = Field(..., description="Distinct calendar years with data for this commodity+state")
    is_best: bool = Field(..., description="True for top-2 months by median price (only when years_of_data >= 3)")
    is_worst: bool = Field(..., description="True for bottom-1 month by median price (only when years_of_data >= 3)")
    month_rank: int = Field(..., description="1=highest median price, 12=lowest")
    model_config = ConfigDict(from_attributes=True)

class SeasonalCalendarResponse(BaseModel):
    commodity: str
    state: str
    months: list[MonthlyStatPoint]
    years_of_data: int = Field(..., description="Max years of data across all months for this selection")
    low_confidence: bool = Field(..., description="True when years_of_data < 3 — frontend must display warning")
    model_config = ConfigDict(from_attributes=True)
```

Create `backend/app/seasonal/routes.py`:

```python
from fastapi import APIRouter, Depends, HTTPException, status
from sqlalchemy.orm import Session
from sqlalchemy import text
from app.database.session import get_db
from app.seasonal.schemas import MonthlyStatPoint, SeasonalCalendarResponse

router = APIRouter(prefix="/seasonal", tags=["Seasonal"])

@router.get(
    "",
    response_model=SeasonalCalendarResponse,
    status_code=status.HTTP_200_OK,
    summary="Get Seasonal Price Calendar",
    description="Returns pre-aggregated monthly price statistics for a commodity in a state. Reads only from seasonal_price_stats — never scans price_history.",
)
def get_seasonal_calendar(
    commodity: str,
    state: str,
    db: Session = Depends(get_db),
) -> SeasonalCalendarResponse:
    rows = db.execute(
        text("""
            SELECT month, median_price, q1_price, q3_price,
                   iqr_price, record_count, years_of_data,
                   is_best, is_worst, month_rank
            FROM seasonal_price_stats
            WHERE commodity_name = :commodity
              AND state_name     = :state
            ORDER BY month ASC
        """),
        {"commodity": commodity, "state": state},
    ).fetchall()

    if not rows:
        raise HTTPException(
            status_code=status.HTTP_404_NOT_FOUND,
            detail=f"No seasonal data for commodity '{commodity}' in state '{state}'. "
                   f"The seasonal calendar may not cover this selection.",
        )

    years_of_data = max(r.years_of_data for r in rows)
    return SeasonalCalendarResponse(
        commodity=commodity,
        state=state,
        months=[MonthlyStatPoint(**dict(r._mapping)) for r in rows],
        years_of_data=years_of_data,
        low_confidence=years_of_data < 3,
    )
```

Note: Use query parameters (`?commodity=Onion&state=Maharashtra`) not path params — avoids URL-encoding issues with state names containing spaces (per research open question #1).
