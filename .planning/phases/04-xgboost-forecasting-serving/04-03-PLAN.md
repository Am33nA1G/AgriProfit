    ```python
    import threading, joblib
    from pathlib import Path
    from cachetools import LRUCache

    # Resolve repo root: this file is at backend/app/ml/loader.py
    REPO_ROOT = Path(__file__).resolve().parent.parent.parent.parent
    ARTIFACTS_DIR = REPO_ROOT / "ml" / "artifacts"

    _lock = threading.Lock()
    # maxsize configurable — import settings lazily to avoid circular imports
    _model_cache: LRUCache = LRUCache(maxsize=20)

    def get_or_load_model(commodity_slug: str):
        """Lazy-load model from cache or disk. Thread-safe.

        Returns:
            ForecasterRecursiveMultiSeries instance, or None if artifact missing.
        Note:
            Use `def` route handlers (not `async def`) when calling this function
            to avoid blocking the event loop (see MEMORY.md OSRM async trap).
        """
        with _lock:
            if commodity_slug in _model_cache:
                return _model_cache[commodity_slug]
        artifact_path = ARTIFACTS_DIR / f"{commodity_slug}.joblib"
        if not artifact_path.exists():
            return None
        model = joblib.load(artifact_path)  # CPU-bound I/O — runs in threadpool via def route
        with _lock:
            _model_cache[commodity_slug] = model
        return model

    def get_model_cache() -> LRUCache:
        """Return cache reference for app.state attachment and monitoring."""
        return _model_cache
    ```

    ```python
    MIN_DAYS_SERVE = 365   # FORE-05: below this -> seasonal fallback

    def mape_to_confidence_colour(mape: float | None) -> str:
        if mape is None: return "Red"
        if mape < 0.10: return "Green"
        if mape < 0.25: return "Yellow"
        return "Red"

    class ForecastService:
        def __init__(self, db: Session):
            self.db = db

        def get_forecast(self, commodity: str, district: str, horizon: int = 14) -> ForecastResponse:
            """
            1. Check forecast_cache for today's cached entry (cache hit -> return immediately)
            2. Check coverage_days for this commodity+district pair in price_history
            3. If coverage_days < MIN_DAYS_SERVE -> return seasonal fallback response
            4. Load model via get_or_load_model(commodity_slug)
            5. If model None -> return seasonal fallback
            6. Call forecaster.predict_interval(steps=horizon, levels=[district_key], interval=[10,90], n_boot=100)
            7. Compute direction, build ForecastResponse, write to forecast_cache, return
            """
            # Cache lookup: query ForecastCache where commodity_name=commodity AND district_name=district
            # AND generated_date=date.today() AND forecast_horizon_days=horizon AND expires_at > now()
            cached = self._lookup_cache(commodity, district, horizon)
            if cached:
                return self._cache_to_response(cached)

            coverage_days = self._get_coverage_days(commodity, district)
            if coverage_days < MIN_DAYS_SERVE:
                return self._seasonal_fallback(commodity, district, horizon)

            slug = commodity.lower().replace(" ", "_").replace("/", "_")
            model = get_or_load_model(slug)
            if model is None:
                return self._seasonal_fallback(commodity, district, horizon)

            return self._invoke_model(model, commodity, district, horizon)

        def _lookup_cache(self, commodity, district, horizon):

        def _seasonal_fallback(self, commodity, district, horizon) -> ForecastResponse:
            return ForecastResponse(

                direction="flat", price_low=None, price_mid=None, price_high=None,
                confidence_colour="Red", tier_label="seasonal average fallback",
                last_data_date="2025-10-30",
                coverage_message=f"Insufficient price history for {district}. Showing seasonal averages.",
            )
    ```
