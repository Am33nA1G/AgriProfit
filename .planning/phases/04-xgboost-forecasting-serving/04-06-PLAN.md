    ```typescript
    // ---------------------------------------------------------------------------
    // Helpers
    // ---------------------------------------------------------------------------

    /**
     * Navigate to /forecast and wait for the page to be interactive.
     * The commodity select is populated via an API call; wait for it to have options.
     */
    async function gotoForecastPage(page: Page): Promise<void> {
      await page.goto('/forecast');
      // Wait for the commodity select to appear (always present in DOM)
      await page.waitForSelector('#commodity-select', { state: 'visible' });
    }

    /**
     * Select a value from a native <select> element by its CSS id.
     */
    async function selectOption(page: Page, selectId: string, value: string): Promise<void> {
      await page.selectOption(`#${selectId}`, value);
    }

    // ---------------------------------------------------------------------------
    // Gap 1: Forecast Page Visual Rendering
    // Verify chart, direction badge, confidence badge, and data freshness note
    // render for a commodity+district pair that returns a full ML forecast.
    // ---------------------------------------------------------------------------
    test('Gap 1: forecast page renders chart and badges for Wheat + Pune', async ({ page }) => {
      // Mock the API response so the test is deterministic and does not require
      // a live ML model. The mock returns a full-model response with all fields.
      await page.route('**/api/v1/forecast/**', async (route) => {
        const mockResponse = {
          commodity: 'wheat',

            { date: '2025-11-01', price_low: 1820, price_mid: 2020, price_high: 2220 },
            { date: '2025-11-02', price_low: 1830, price_mid: 2030, price_high: 2230 },
            { date: '2025-11-03', price_low: 1840, price_mid: 2040, price_high: 2240 },
            { date: '2025-11-04', price_low: 1850, price_mid: 2050, price_high: 2250 },
            { date: '2025-11-05', price_low: 1860, price_mid: 2060, price_high: 2260 },
            { date: '2025-11-06', price_low: 1870, price_mid: 2070, price_high: 2270 },
            { date: '2025-11-07', price_low: 1880, price_mid: 2080, price_high: 2280 },
            { date: '2025-11-08', price_low: 1890, price_mid: 2090, price_high: 2290 },
            { date: '2025-11-09', price_low: 1900, price_mid: 2100, price_high: 2300 },
            { date: '2025-11-10', price_low: 1910, price_mid: 2110, price_high: 2310 },
            { date: '2025-11-11', price_low: 1920, price_mid: 2120, price_high: 2320 },
            { date: '2025-11-12', price_low: 1930, price_mid: 2130, price_high: 2330 },
            { date: '2025-11-13', price_low: 1940, price_mid: 2140, price_high: 2340 },
          ],
        };
        await route.fulfill({
          status: 200,
          contentType: 'application/json',
          body: JSON.stringify(mockResponse),

      });

    test('Gap 2: seasonal fallback banner renders for low-coverage district', async ({ page }) => {
      // Mock API to return a seasonal fallback response
      await page.route('**/api/v1/forecast/**', async (route) => {
        const url = route.request().url();
        // Only intercept the forecast endpoint, not commodities
        if (url.includes('/forecast/commodities')) {
          await route.continue();
          return;
        }
        const fallbackResponse = {
          commodity: 'jute',
          district: 'Bellary',
          horizon_days: 14,
          direction: 'flat',
          price_low: null,
          price_mid: null,
          price_high: null,
          confidence_colour: 'Yellow',
          tier_label: 'seasonal average fallback',
          last_data_date: '2025-06-15',
          r2_score: null,
          coverage_message: 'Insufficient price history for Bellary. Showing seasonal averages.',
          forecast_points: [],
        };
        await route.fulfill({
          status: 200,
          contentType: 'application/json',
          body: JSON.stringify(fallbackResponse),
        });

      await page.route('**/api/v1/forecast/commodities', async (route) => {
        await route.fulfill({
          status: 200,
          contentType: 'application/json',
          body: JSON.stringify(['jute', 'wheat', 'rice']),
        });
      });

      await gotoForecastPage(page);

      // Select a commodity that triggers fallback
      await selectOption(page, 'commodity-select', 'jute');
      await selectOption(page, 'state-select', 'Karnataka');
      await selectOption(page, 'district-select', 'Bellary');

      // Wait for forecast-result to appear (the API mock returns quickly)
      await page.waitForSelector('#forecast-result', { state: 'visible', timeout: 10_000 });

      // 1. Fallback banner must be visible
      const fallbackBanner = page.locator('#fallback-banner');
      await expect(fallbackBanner).toBeVisible();

      // 2. Banner contains "Limited Data Coverage" heading
      await expect(fallbackBanner).toContainText('Limited Data Coverage');

      // 3. Banner contains the coverage_message text
      await expect(fallbackBanner).toContainText('Insufficient price history for Bellary');

      // 4. Tier badge shows "Seasonal Average" (tier_label != "full model")
      const badges = page.locator('#forecast-badges');
      await expect(badges).toContainText('Seasonal Average');

      // 5. Direction badge shows "Stable" (direction=flat)
      await expect(badges).toContainText('Stable');

      // 6. Price range card is NOT rendered (price_mid is null → falsy check in page.tsx)
      await expect(page.locator('#price-range-card')).not.toBeVisible();

      // 7. Data freshness note still shown (last_data_date always rendered when forecast present)
      await expect(page.locator('#data-freshness')).toBeVisible();
      await expect(page.locator('#data-freshness')).toContainText('Data last updated: 2025-06-15');
    });

    // ---------------------------------------------------------------------------
    // Gap 3: Cascading Select Reset
    // Verify that changing state clears the district dropdown and prevents the
    // forecast query from firing for the old district.
    // ---------------------------------------------------------------------------
    test('Gap 3: changing state clears district and does not fire stale forecast query', async ({ page }) => {
      // Track forecast API requests to verify stale query does not fire
      const forecastRequests: string[] = [];

      await page.route('**/api/v1/forecast/commodities', async (route) => {
        await route.fulfill({
          status: 200,
          contentType: 'application/json',
          body: JSON.stringify(['wheat', 'rice', 'tomato']),
        });
      });

      await page.route('**/api/v1/forecast/**', async (route) => {
        const url = route.request().url();
        if (!url.includes('/commodities')) {
          forecastRequests.push(url);
        }
        // Abort the request — we only care about whether it fired, not the response
        await route.abort();
      });

      await gotoForecastPage(page);

      // Step 1: Select commodity + Maharashtra + Pune
      await selectOption(page, 'commodity-select', 'wheat');
      await selectOption(page, 'state-select', 'Maharashtra');
      await selectOption(page, 'district-select', 'Pune');

      // At this point one forecast request fires (wheat + Pune)
      // Wait briefly for it to be intercepted
      await page.waitForTimeout(500);
      const requestsAfterFirstSelect = forecastRequests.length;
      // At least one request should have fired (wheat/Pune)
      expect(requestsAfterFirstSelect).toBeGreaterThanOrEqual(1);
      const puneRequestExists = forecastRequests.some((url) => url.includes('Pune'));
      expect(puneRequestExists).toBe(true);

      // Step 2: Change state to Karnataka — this triggers setState("Karnataka") + setDistrict("")
      forecastRequests.length = 0; // reset tracker
      await selectOption(page, 'state-select', 'Karnataka');

      // Wait briefly — no forecast request should fire because district is now ""
      await page.waitForTimeout(500);

      // 3. District dropdown value must be reset (empty string = "Select District" option selected)
      const districtValue = await page.inputValue('#district-select');
      expect(districtValue).toBe('');

      // 4. District dropdown options must now show Karnataka's districts (not Maharashtra's)
      const districtOptions = await page.locator('#district-select option').allTextContents();
      expect(districtOptions).toContain('Bangalore');
      expect(districtOptions).not.toContain('Pune');

      // 5. No forecast request fired for the old Pune combination
      // (canFetch = commodity && district — district is "", so enabled=false in useQuery)
      const puneStaleRequest = forecastRequests.some((url) => url.includes('Pune'));
      expect(puneStaleRequest).toBe(false);

      // 6. No forecast request at all fired after state change (district is empty)
      expect(forecastRequests.length).toBe(0);

      // 7. Page shows empty state (forecast-empty visible when !canFetch)
      // canFetch = commodity && district — commodity is set but district is "" → falsy
      await expect(page.locator('#forecast-empty')).toBeVisible();
    });
    ```

    Note on the route mock ordering: `page.route('**/api/v1/forecast/**')` will also match
    `/forecast/commodities`. In Gap 3, commodities is mocked first with a more specific pattern.
    In Playwright, routes are matched in registration order — the commodities route registered first
    intercepts the commodities request before the wildcard catches it. This is the correct ordering.
