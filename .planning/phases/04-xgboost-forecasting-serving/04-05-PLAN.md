    ```typescript
    import api from "@/lib/api";  // existing axios client with baseURL including /api/v1

    export interface ForecastPoint {
      date: string;
      price_low?: number;
      price_mid?: number;
      price_high?: number;
    }

    export interface ForecastResponse {
      commodity: string;
      district: string;
      horizon_days: number;
      direction: "up" | "down" | "flat";
      price_low: number | null;
      price_mid: number | null;
      price_high: number | null;
      confidence_colour: "Green" | "Yellow" | "Red";
      tier_label: "full model" | "seasonal average fallback";
      last_data_date: string;
      forecast_points: ForecastPoint[];
      coverage_message: string | null;
    }

    export const forecastService = {
      getForecast: async (commodity: string, district: string, horizon: number = 14): Promise<ForecastResponse> => {
        const encodedCommodity = encodeURIComponent(commodity);
        const encodedDistrict = encodeURIComponent(district);
        const { data } = await api.get<ForecastResponse>(
          `/forecast/${encodedCommodity}/${encodedDistrict}?horizon=${horizon}`
        );
        return data;
      },
    };
    ```

    Create frontend/src/components/ForecastChart.tsx:
    ```tsx
    "use client"

    import {
      ComposedChart, Line, Area, XAxis, YAxis,
      CartesianGrid, Tooltip, Legend, ResponsiveContainer,
    } from "recharts";
    import type { ForecastPoint } from "@/services/forecast";

    interface ForecastChartProps {
      points: ForecastPoint[];
      lastDataDate: string;  // date string — visual separator between history and forecast
    }

    export function ForecastChart({ points, lastDataDate }: ForecastChartProps) {
      // Chart data: historical entries have price_mid (actual), forecast entries have low/mid/high band
      // Visual separator: dashed vertical line at lastDataDate (use ReferenceLine from recharts)
      return (
        <ResponsiveContainer width="100%" height={300}>
          <ComposedChart data={points} margin={{ top: 8, right: 16, bottom: 0, left: 0 }}>
            <CartesianGrid strokeDasharray="3 3" stroke="#f0f0f0" />
            <XAxis dataKey="date" tick={{ fontSize: 11 }} />
            <YAxis tick={{ fontSize: 11 }} unit=" ₹" />
            <Tooltip formatter={(value: number) => [`₹${value.toFixed(0)}`, ""]} />
            <Legend />
            {/* Confidence band: high area filled, then low area covers it in white */}
            <Area type="monotone" dataKey="price_high" name="High" stroke="none" fill="#dbeafe" fillOpacity={0.6} />
            <Area type="monotone" dataKey="price_low" name="Low" stroke="none" fill="#ffffff" fillOpacity={1} />
            {/* Mid forecast line (dashed blue) */}
            <Line type="monotone" dataKey="price_mid" name="Forecast" stroke="#2563eb" strokeDasharray="5 5" dot={false} strokeWidth={2} />
          </ComposedChart>
        </ResponsiveContainer>
      );
    }
    ```

    Note: The Area layering trick (high area filled blue, low area filled white on top) creates the visual confidence band without needing a custom shape. This is the pattern from the research.
