    ```python
        """Query price_history for commodity, pivot to wide format, filter by min_days.

        Returns:
            series_df: pd.DataFrame with DatetimeIndex, columns=district_canonical_name
            excluded: list[dict] with {"district": str, "reason": str}
        """

    def train_commodity(commodity, series_df, excluded, db):
        """Full train+validate+log+save cycle for one commodity."""
        forecaster = ForecasterRecursiveMultiSeries(
            regressor=XGBRegressor(n_estimators=200, max_depth=6, learning_rate=0.05, random_state=42),
            lags=[7, 14, 30, 90],
            encoding='ordinal',
            store_in_sample_residuals=True,  # required for predict_interval bootstrapping
        )
        cv = TimeSeriesFold(n_splits=4, steps=14,
                            initial_train_size=int(len(series_df) * 0.7),
                            refit=False, fixed_train_size=False)
        metrics_df, _ = backtesting_forecaster_multiseries(
            forecaster=forecaster, series=series_df, cv=cv,
            metric=['mean_squared_error', 'mean_absolute_percentage_error'],
            add_aggregated_metric=True, n_jobs='auto',
        )
        rmse_arr = np.sqrt(metrics_df['mean_squared_error'].values)
        mape_arr = metrics_df['mean_absolute_percentage_error'].values
        slug = commodity.lower().replace(" ", "_").replace("/", "_")
        artifact_path = str(ARTIFACTS_DIR / f"{slug}.joblib")
        # GATE: log first, write artifact only if log succeeds
        log_training(db, commodity, n_series=len(series_df.columns),
                     n_folds=4, rmse_arr=rmse_arr, mape_arr=mape_arr,
                     artifact_path=artifact_path, excluded=excluded)
        forecaster.fit(series=series_df)
        joblib.dump(forecaster, artifact_path)
        print(f"[OK] {commodity}: RMSE={rmse_arr.mean():.2f} MAPE={mape_arr.mean():.4f} -> {artifact_path}")
    ```
